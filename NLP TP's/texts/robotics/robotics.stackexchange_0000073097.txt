A:

Does this help?
namespace Choices
{
  enum Choice
  {
     Choice1,
     Choice2,
     Choice3
  };
}
typedef Choices::Choice Choice;

int foo() {
    Choice c = Choices::Choice1;

    switch(c) {
        case Choices::Choice1:
            // TODO: something
            break;
        case Choices::Choice2:
            // TODO: something
            break;
        case Choices::Choice3:
            // TODO: something
            break;
        default:
            // TODO: something
            break;
    }
} 

(tested and compiles with gcc 4.8.4)
