A:

This is working for me:
include_directories(
# include
  ${catkin_INCLUDE_DIRS}
  /usr/include/CL/
)
add_executable(save_pattern src/save_pattern.cpp)
target_link_libraries(save_pattern ${catkin_LIBRARIES} /usr/lib/x86_64-linux-gnu/libOpenCL.so)
