A:

I used this launch file and  openni.launch in openni_launch
<launch>
<remap from="/camera/rgb/image_rect" to="/camera/rgb/image_color"/>
<remap from="/camera/depth_registered/image_rect" to="/camera/depth_registered/image_raw"/>

<remap from="/camera/depth_registered/camera_info" to="/camera/depth/camera_info"/>

<remap from="/camera/depth_registered/image_rect" to="/camera/depth/image"/>

<node
    pkg="fovis_ros"
    type="fovis_mono_depth_odometer"
    name="fovis"
    output="screen"
        
/>

</launch>
