A:

Other symptoms include:

Not all the cores on the computer are
working. Testable with: cat/proc/cpuinfo

The following code has   4 ms intervals
 #include <sys/time.h>
 #include <time.h>
 #include <stdio.h>

 main()
 {
   struct timeval tv;
   struct timeval tv_old;
   gettimeofday(&tv, NULL);
   tv_old = tv;
   while(1)
   {
     gettimeofday(&tv, NULL);
     if (tv_old.tv_usec != tv.tv_usec)
     {
       printf("%ld %ld\n", tv.tv_sec, tv.tv_usec);
       tv_old = tv;

       // sleep 1 nano second
       struct timespec nsleep;
       nsleep.tv_sec = 0;
       nsleep.tv_nsec = 1;
       nanosleep(&nsleep, NULL);
     }
   }
 }

##Solution

Edit /etc/default/grub and remove acpi=off
sudo update-grub
restart
