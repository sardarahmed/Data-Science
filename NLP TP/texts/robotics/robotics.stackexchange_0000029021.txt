A:

This is not ideal, but one trick is to also add a <script><trajectory> with 2 waypoints that have the same pose. Here's an example:
  <actor name="talkingA">
    <pose>5 5 5.76 0 0 0</pose>
    <skin>
      <filename>model://actor/meshes/SKIN_man_blue_shirt.dae</filename>
    </skin>
    <animation name="animation">
      <filename>model://actor/meshes/ANIMATION_talking_a.dae</filename>
      
    </animation>
      <script>
        <trajectory id="0" type="animation">
          
            <waypoint>
              <time>100</time>
              <pose>5 5 5.16 0 0 0</pose>
            </waypoint>
          
            <waypoint>
              <time>100</time>
              <pose>5 5 5.16 0 0 0</pose>
            </waypoint>
          
        </trajectory>
      </script>
  </actor>
