A:

here is an example:
  plugins:
    - name: pc_layer
    type: 'costmap_2d::ObstacleLayer'

  pc_layer:
    enabled: true
    observation_sources: rgbd_cloud
    combination_method: 1 

    rgbd_cloud:
      sensor_frame: base_footprint
      data_type: PointCloud2
      topic: cloud 
      expected_update_rate: 1.0
      observation_persistence: 0.0
      marking: true
      clearing: true
      raytrace_range: 4.0
      obstacle_range: 2.5
